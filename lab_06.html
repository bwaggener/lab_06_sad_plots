fish_longer <- fish %>%
  pivot_longer(cols = -country, names_to = "type", values_to = "total")

fish_proportion <- fish_longer %>%
  group_by(country) %>%
  mutate(total_country = sum(total[type != "total"])) %>%  
  filter(type != "total") %>%  # Remove 'total' row
  mutate(proportion = total / total_country) %>% 
  ungroup()
